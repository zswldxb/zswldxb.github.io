示例

import math
import torch
import torch.nn as nn

print(f'torch version: {torch.__version__}')
# torch version: 1.10.2

# torch.set_printoptions(sci_mode=False)


def gaussian_nll(input, target, var, reduction, full, eps):
    var = torch.where(var > eps, var, torch.tensor([eps]))
    constant = 0.5 * math.log(2 * math.pi)

    y_hat = 0.5 * (torch.log(var) + torch.div((target - input)**2, var))
    if full:
        y_hat +=constant

    if reduction == 'mean':
        y_hat = torch.mean(y_hat)
    elif reduction == 'sum':
        y_hat = torch.sum(y_hat)

    return y_hat


def solve(features, reduction='mean', full=False, eps=1e-6):
    reduction_list = ['none', 'mean', 'sum']
    assert reduction in reduction_list, 'mode error'

    input = torch.rand(features)
    target = (torch.rand(features) * 5).long().to(torch.float32)
    var = torch.rand(features)

    loss = nn.GaussianNLLLoss(reduction=reduction, full=full, eps=eps)
    y = loss(input, target, var)

    y_hat = gaussian_nll(input, target, var, reduction, full, eps)

    print(f'error: {(torch.abs(y-y_hat)).flatten()}')


if __name__ == '__main__':
    features = (10, 10)

    solve(features, reduction='none', full=False, eps=1e-6)

    solve(features, reduction='mean', full=False, eps=1e-6)

    solve(features, reduction='sum', full=False, eps=1e-6)

    solve(features, reduction='none', full=True, eps=1e-6)
