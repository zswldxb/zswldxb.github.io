Pytorch.torch.nn.MultiLabelMarginLoss

损失函数层 MultiLabelMarginLoss\text{MultiLabelMarginLoss}MultiLabelMarginLoss 为多标签间隔损失函数. 损失函数层 MultiLabelMarginLoss\text{MultiLabelMarginLoss}MultiLabelMarginLoss 其要求标签张量 Y∈RN×CY\in\R^{N\times C}Y∈RN×C, 输入张量 X∈RN×CX\in\R^{N\times C}X∈RN×C. 且 YYY 的每个元素取值集合为 [−1,C)∩N[-1,C)\cap\N[−1,C)∩N, 且每个样本仅 −1-1−1 之间的类别为有效类别.

如标签张量为 [[2, 3, -1, 0], [0, -1, 2, 1]], 则第 000 个样本的有效标签为 2,32, 32,3, 无效标签为 0,10,10,1, 第 111 个样本的有效标签为 000, 无效标签为 1,2,31,2,31,2,3.

reduction 参数

reduction: 用于指定计算的模式, 'none', 'mean' 或 'sum'. 默认为 'mean'.

设 UYtU_{Y_{t}}UYt​​ 集合为第 ttt 个样本的有效标签集合, VYtV_{Y_{t}}VYt​​ 集合为第 ttt 个样本的无效标签集合.

示例

import torch
import torch.nn as nn
import numpy as np
import random

print(f'torch version: {torch.__version__}')
# torch version: 1.10.2

# torch.set_printoptions(sci_mode=False)


def multi_label_margin(input, target, reduction):
    N, C = input.shape
    y_hat = torch.zeros(N)

    for t in range(N):
        u, v = set(), set(range(C))
        for e in target[t].tolist():
            if e == -1:
                break
            u.add(e)
        v = v - u
        for ui in u:
            for vi in v:
                y_hat[t] += max(0, 1 - input[t, ui] + input[t, vi])
    y_hat /= C

    if reduction == 'mean':
        y_hat = torch.mean(y_hat)
    elif reduction == 'sum':
        y_hat = torch.sum(y_hat)

    return y_hat


def solve(batch_size, num_class, reduction='mean'):
    reduction_list = ['none', 'mean', 'sum']
    assert reduction in reduction_list, 'mode error'

    input = torch.rand(batch_size, num_class)
    target = []
    for bs in range(batch_size):
        l = np.arange(-1, num_class)
        random.shuffle(l)
        target.append(l[:-1])
    target = torch.tensor(target, dtype=torch.long)

    loss = nn.MultiLabelMarginLoss(reduction=reduction)
    y = loss(input, target)

    y_hat = multi_label_margin(input, target, reduction)
    # print(y)

    print(f'error: {(torch.abs(y-y_hat)).flatten()}')


if __name__ == '__main__':
    batch_size, num_class = (5, 10)

    solve(batch_size, num_class, reduction='none')

    solve(batch_size, num_class, reduction='mean')

    solve(batch_size, num_class, reduction='sum')
