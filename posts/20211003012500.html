示例

import torch
import torch.nn as nn

print(f'torch version: {torch.__version__}')
# torch version: 1.10.2

# torch.set_printoptions(sci_mode=False)


def dis(u, v, p):
    return torch.sum(torch.abs(u - v)**p, dim=-1)**(1 / p)


def triplet_margin(anchor, positive, negative, reduction, margin, p):
    z = margin + dis(anchor, positive, p) - dis(anchor, negative, p)
    y_hat = torch.maximum(z, torch.zeros_like(z))

    if reduction == 'mean':
        y_hat = torch.mean(y_hat)
    elif reduction == 'sum':
        y_hat = torch.sum(y_hat)

    return y_hat


def solve(features, reduction='mean', margin=1, p=2):
    reduction_list = ['none', 'mean', 'sum']
    assert reduction in reduction_list, 'mode error'

    anchor = torch.rand(features)
    positive = torch.rand(features)
    negative = torch.rand(features)

    loss = nn.TripletMarginLoss(reduction=reduction, margin=margin, p=p)
    y = loss(anchor, positive, negative)

    y_hat = triplet_margin(anchor, positive, negative, reduction, margin, p)

    print(f'error: {(torch.abs(y-y_hat)).flatten()}')


if __name__ == '__main__':
    features = (5, 10, 15)

    solve(features, reduction='none', p=1)

    solve(features, reduction='mean', p=1)

    solve(features, reduction='sum', p=1)

    solve(features, reduction='none', p=2)

    solve(features, reduction='mean', p=2)

    solve(features, reduction='sum', p=2)
