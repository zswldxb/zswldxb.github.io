Pytorch.torch.nn.BCELoss

损失函数层 BCELoss\text{BCELoss}BCELoss 为二分类交叉熵损失函数. 损失函数层 BCELoss\text{BCELoss}BCELoss 其要求标签张量 YYY, 输入张量 XXX 均为 nnn 维张量, 且 X,Y∈Rd1×⋯×dnX,Y\in\R^{d_{1}\times\cdots\times d_{n}}X,Y∈Rd1​×⋯×dn​. 且 YYY 的每个元素取值集合为 {0,1}\{0,1\}{0,1}, XXX 的每个元素的取值范围为 [0,1][0,1][0,1].

对于标签为 yyy, 预测为 xxx, 二分类交叉熵定义如下:

l(x,y)=−(ylog⁡(x)+(1−y)log⁡(1−x))l(x,y)=-(y\log {(x)}+(1-y)\log{(1-x)})l(x,y)=−(ylog(x)+(1−y)log(1−x))

weight 参数

weight 参数用于加权每个元素的权重. W∈Rd1×⋯×dnW\in\R^{d_{1}\times\cdots\times d_{n}}W∈Rd1​×⋯×dn​.

reduction 参数

reduction: 用于指定计算的模式, 'none', 'mean', 'sum' 或 'batchmean'. 默认为 'mean'. 以后 'mean' 的计算过程将与 'batchmean' 相同.

示例

import torch
import torch.nn as nn

print(f'torch version: {torch.__version__}')
# torch version: 1.10.2

# torch.set_printoptions(sci_mode=False)


def bce(input, target, weight, reduction):
    y_hat = -weight * (target * torch.clamp(torch.log(input), -100) +
                       (1 - target) * torch.clamp(torch.log(1 - input), -100))

    if reduction == 'mean':
        y_hat = torch.mean(y_hat)
    elif reduction == 'sum':
        y_hat = torch.sum(y_hat)

    return y_hat


def solve(features, reduction='mean'):
    reduction_list = ['none', 'mean', 'sum']
    assert reduction in reduction_list, 'mode error'

    input = torch.rand(features)
    target = (torch.rand(features) * 2).long().to(torch.float32)
    weight = torch.rand(features)

    loss = nn.BCELoss(weight=weight, reduction=reduction)
    y = loss(input, target)

    y_hat = bce(input, target, weight, reduction)

    print(f'error: {(torch.abs(y-y_hat)).flatten()}')


if __name__ == '__main__':
    features = (5, 10, 10)

    solve(features, reduction='none')

    solve(features, reduction='mean')

    solve(features, reduction='sum')
