Pytorch.torch.nn.MultiMarginLoss

损失函数层 MultiMarginLoss\text{MultiMarginLoss}MultiMarginLoss 为多分类间隔损失函数. 损失函数层 MultiMarginLoss\text{MultiMarginLoss}MultiMarginLoss 其要求标签张量 Y∈RNY\in\R^{N}Y∈RN, 输入张量 X∈RN×CX\in\R^{N\times C}X∈RN×C. 且 YYY 的每个元素取值集合为 [0,C)∩N[0,C)\cap\N[0,C)∩N.

p 参数

p 参数用于指定使用 ppp 指数计算. p∈{1,2}p\in\{1,2\}p∈{1,2}

margin 参数

margin 参数用于表示间隔.

weight 参数

weight 参数用于加权每个样本的权重. W∈RCW\in\R^{C}W∈RC.

reduction 参数

reduction: 用于指定计算的模式, 'none', 'mean' 或 'sum'. 默认为 'mean'.

当使用 ppp 指数, 间隔为 margin\text{margin}margin, 权重为 WWW 时.

当 reduction='none' 时, 计算过程如下:

示例

import torch
import torch.nn as nn

print(f'torch version: {torch.__version__}')
# torch version: 1.10.2

# torch.set_printoptions(sci_mode=False)


def multi_margin(input, target, reduction, p, margin, weight):
    target = target.unsqueeze(1)
    weight = weight.unsqueeze(0).expand(input.shape)
    w = torch.gather(weight, dim=1, index=target)
    u = torch.gather(input, dim=1, index=target)
    v = torch.maximum(torch.zeros_like(input), w * (margin - u + input))
    mask = torch.ones_like(input, dtype=torch.long).scatter_(dim=1, index=target, value=0)
    y_hat = torch.mean((v * mask) ** p, dim=1)

    if reduction == 'mean':
        y_hat = torch.mean(y_hat)
    elif reduction == 'sum':
        y_hat = torch.sum(y_hat)

    return y_hat


def solve(batch_size, num_class, reduction='mean', p=1, margin=1, use_weight=False):
    reduction_list = ['none', 'mean', 'sum']
    assert reduction in reduction_list, 'mode error'

    input = torch.rand(batch_size, num_class)
    target = (torch.rand(batch_size) * num_class).long()
    if use_weight:
        weight = torch.rand(num_class)
    else:
        weight = torch.ones(num_class)

    loss = nn.MultiMarginLoss(reduction=reduction, p=p, margin=margin, weight=weight)
    y = loss(input, target)

    y_hat = multi_margin(input, target, reduction, p, margin, weight)

    print(f'error: {(torch.abs(y-y_hat)).flatten()}')


if __name__ == '__main__':
    batch_size, num_class = (20, 20)

    solve(batch_size, num_class, reduction='none', p=1)

    solve(batch_size, num_class, reduction='mean', p=1)

    solve(batch_size, num_class, reduction='sum', p=1)

    solve(batch_size, num_class, reduction='mean', p=2)

    solve(batch_size, num_class, reduction='sum', p=2, margin=3)

    solve(batch_size, num_class, reduction='none', p=1, use_weight=True)

    solve(batch_size, num_class, reduction='mean', p=1, use_weight=True)

    solve(batch_size, num_class, reduction='sum', p=1, use_weight=True)
