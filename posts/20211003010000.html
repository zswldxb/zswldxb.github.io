示例

import torch
import torch.nn as nn
import numpy as np

print(f'torch version: {torch.__version__}')
# torch version: 1.10.2

def solve(input, batch_size, channels, features, groups):
    layer = nn.GroupNorm(groups, channels)
    y = layer(torch.tensor(input)).detach().numpy()

    # print(f'output.shape: {y}')

    eps = 1e-05
    if isinstance(features, int):
        features = [features, ]
    d_0 = batch_size * groups
    weight = layer.weight.detach().numpy().reshape(1, channels, *[1 for _ in features])
    bias = layer.bias.detach().numpy().reshape(1, channels, *[1 for _ in features])
    x = input.reshape((batch_size, groups, channels // groups, *features)).reshape(d_0, -1)
    mu = np.mean(x, axis=1, keepdims=True)
    nu = np.mean((x-mu)**2, axis=1, keepdims=True)
    x_hat = ((x - mu) / ((nu + eps) ** 0.5)).reshape(input.shape)
    y_hat = weight * x_hat + bias
    print(f'output error: {np.mean(np.abs((y-y_hat).flatten()))}')


if __name__ == '__main__':
    batch_size = 2
    channels = 6
    features = (2, 5, 4, 5, 6)
    groups = 2
    input = np.arange(batch_size * channels * np.prod(features), dtype=np.float32).reshape((batch_size, channels, *features))
    # print(f'input.shape: {input}')
    solve(input, batch_size, channels, features, groups)
