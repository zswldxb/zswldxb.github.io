margin 参数

margin 参数表示间隔, 用于区分正类与负类之间的距离.

reduction 参数

reduction: 用于指定计算的模式, 'none', 'mean', 'sum' 或 'batchmean'. 默认为 'mean'.

当 reduction='none' 时, 计算过程如下:

示例

import torch
import torch.nn as nn

print(f'torch version: {torch.__version__}')
# torch version: 1.10.2

# torch.set_printoptions(sci_mode=False)

def cos_similarity(input1, input2):
    # input1, input2: [N, D]
    input1 = input1 / torch.sqrt(torch.sum(input1 ** 2, dim=-1, keepdim=True))
    input2 = input2 / torch.sqrt(torch.sum(input2 ** 2, dim=-1, keepdim=True))
    cos_sim = torch.sum(input1 * input2, dim=-1)
    return cos_sim

def cosine_embedding(input1, input2, target, reduction, margin):
    cos_sim = cos_similarity(input1, input2)
    z1 = 1 - cos_sim
    z2 = cos_sim - margin
    z2 = torch.where(z2 > 0, z2, torch.zeros_like(z2))
    y_hat = torch.where(target == 1, z1, z2)

    if reduction == 'mean':
        y_hat = torch.mean(y_hat)
    elif reduction == 'sum':
        y_hat = torch.sum(y_hat)

    return y_hat


def solve(batch_size, num_class, reduction='mean', margin=0.5):
    reduction_list = ['none', 'mean', 'sum']
    assert reduction in reduction_list, 'mode error'

    input1 = torch.rand(batch_size, num_class)
    input2 = torch.rand(batch_size, num_class)
    target = torch.where(torch.rand(batch_size) > 0.5, 1, -1)

    loss = nn.CosineEmbeddingLoss(margin=margin, reduction=reduction)
    y = loss(input1, input2, target)
    
    y_hat = cosine_embedding(input1, input2, target, reduction, margin)

    print(f'error: {(torch.abs(y-y_hat)).flatten()}')


if __name__ == '__main__':
    batch_size, num_class = 50, 20

    solve(batch_size, num_class, reduction='none', margin=0.5)

    solve(batch_size, num_class, reduction='mean', margin=0.5)

    solve(batch_size, num_class, reduction='sum', margin=0.5)
