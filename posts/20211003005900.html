LayerNorm\text{LayerNorm}LayerNorm 网络层的训练过程与测试过程相同. 不采用动量计算测试时的均值和方差.

示例

import torch
import torch.nn as nn
import numpy as np

print(f'torch version: {torch.__version__}')
# torch version: 1.10.2

def solve(input, normalized_shape):
    layer = nn.LayerNorm(normalized_shape)
    y = layer(torch.tensor(input)).detach().numpy()
    # print(f'output.shape: {y}')

    if isinstance(normalized_shape, int):
        normalized_shape = [normalized_shape, ]
    orig_shape = input.shape
    eps = 1e-05
    weight = layer.weight.detach().numpy()
    bias = layer.bias.detach().numpy()
    x = np.copy(input).reshape(-1, np.prod(normalized_shape))
    mu = np.mean(x, axis=1, keepdims=True)
    nu = np.mean((x-mu)**2, axis=1, keepdims=True)
    x_hat = ((x - mu) / ((nu + eps) ** 0.5)).reshape(orig_shape)
    y_hat = weight * x_hat + bias
    print(f'output error: {np.mean(np.abs((y-y_hat).flatten()))}')


if __name__ == '__main__':
    # NLP Example
    batch, sentence_length, embedding_dim = 20, 5, 10
    embedding = np.random.randn(batch, sentence_length, embedding_dim).astype(np.float32)
    solve(embedding, embedding_dim)

    # Image Example
    N, C, H, W = 20, 5, 10, 10
    input = np.random.randn(N, C, H, W).astype(np.float32)
    solve(input, [C, H, W])
