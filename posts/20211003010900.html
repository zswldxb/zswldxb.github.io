示例

import torch
import torch.nn as nn

print(f'torch version: {torch.__version__}')
# torch version: 1.10.2

# torch.set_printoptions(sci_mode=False)


def cross_entropy(input, target, weight):
    y_hat = -(torch.gather(weight * input, dim=1, index=target.unsqueeze(1))).squeeze(1)
    return y_hat

def solve(batch_size, channels, features, use_weight=False, mode='mean', ignore_index=-100):
    mode_list = ['none', 'mean', 'sum']
    assert mode in mode_list, 'mode error'

    input = torch.rand((batch_size, channels, *features))
    weight = torch.rand(channels) if use_weight else torch.ones(channels)
    target = (torch.rand((batch_size, *features)) * channels).long()
    one_hot = torch.zeros(batch_size, channels, *features).scatter_(dim=1, index=target.unsqueeze(1), value=1)

    if 0 <= ignore_index and ignore_index < channels: # code for ignore_index
        weight[ignore_index] = 0.0

    loss = nn.NLLLoss(reduction=mode, weight=weight, ignore_index=ignore_index)
    y = loss(input, target)

    weight = weight.reshape(1, channels, *[1 for _ in range(len(features))])
    y_hat = cross_entropy(input, target, weight)

    # print(y)
    # print(input)
    if mode == 'mean':
        y_hat = torch.sum(y_hat) / torch.sum(weight * one_hot)
    elif mode == 'sum':
        y_hat = torch.sum(y_hat)

    print(f'error: {(torch.abs(y-y_hat)).flatten()}')


if __name__ == '__main__':
    batch_size = 2
    channels = 8
    features = (50, 50)
    # solve(batch_size, channels, features, mode='none')

    solve(batch_size, channels, features, use_weight=True, mode='mean')

    solve(batch_size, channels, features, use_weight=True, mode='sum')

    solve(batch_size, channels, features, ignore_index=0, mode='none')

    solve(batch_size, channels, features, ignore_index=0, mode='mean')

    solve(batch_size, channels, features, ignore_index=0, mode='sum')
