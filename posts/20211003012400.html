示例

import torch
import torch.nn as nn

print(f'torch version: {torch.__version__}')
# torch version: 1.10.2

# torch.set_printoptions(sci_mode=False)


def multi_label_soft_margin(input, target, reduction, weight):
    def sigmoid(x):
        return 1 / (1 + torch.exp(-x))

    y_hat = -torch.mean(weight * (target * torch.log(sigmoid(input)) + (1 - target) * torch.log(1 - sigmoid(input))), dim=1)

    if reduction == 'mean':
        y_hat = torch.mean(y_hat)
    elif reduction == 'sum':
        y_hat = torch.sum(y_hat)

    return y_hat


def solve(batch_size, num_class, features, reduction='mean', use_weight=False):
    reduction_list = ['none', 'mean', 'sum']
    assert reduction in reduction_list, 'mode error'

    input = torch.rand((batch_size, num_class, *features))
    target = (torch.rand((batch_size, num_class, *features)) * 2).long()
    if use_weight:
        weight = torch.rand(num_class)
    else:
        weight = torch.ones(num_class)
    weight = weight.reshape(1, num_class, *[1 for _ in features])

    loss = nn.MultiLabelSoftMarginLoss(reduction=reduction, weight=weight)
    y = loss(input, target)

    y_hat = multi_label_soft_margin(input, target, reduction, weight)

    print(f'error: {(torch.abs(y-y_hat)).flatten()}')


if __name__ == '__main__':
    batch_size = 5
    num_class = 10
    features = (5, 10, 5)

    solve(batch_size, num_class, features, reduction='none', use_weight=False)

    solve(batch_size, num_class, features, reduction='mean', use_weight=False)

    solve(batch_size, num_class, features, reduction='sum', use_weight=False)

    solve(batch_size, num_class, features, reduction='none', use_weight=True)

    solve(batch_size, num_class, features, reduction='mean', use_weight=True)

    solve(batch_size, num_class, features, reduction='sum', use_weight=True)
