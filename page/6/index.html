import torch
import torch.nn as nn

print(f'torch version: {torch.__version__}')
# torch version: 1.10.2


def solve(batch_size, channels, height, width, mode='mean'):
    mode_list = ['none', 'mean', 'sum']
    assert mode in mode_list, 'mode error'
    input = torch.rand((batch_size, channels, height, width))
    target = torch.ones((batch_size, channels, height, width))

    loss = nn.MSELoss(reduction=mode)
    y = loss(input, target)
    if mode == 'none':
        y_hat = (input - target)**2
    elif mode == 'mean':
        y_hat = torch.mean((input - target)**2)
    else:
        y_hat = torch.sum((input - target)**2)

    print(f'error: {(y-y_hat).flatten()}')
